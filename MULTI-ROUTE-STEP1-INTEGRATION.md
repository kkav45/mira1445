### 2. Новая функция analyzeAllRoutes()

```javascript
MultiRouteUI.analyzeAllRoutes = async function() {
    // 1. Проверка: есть ли точки взлёта
    if (MultiRouteModule.takeoffPoints.length === 0) {
        alert('Добавьте хотя бы одну точку взлёта!');
        return;
    }

    // 2. Проверка: есть ли маршруты
    if (MultiRouteModule.routes.length === 0) {
        alert('Загрузите хотя бы один маршрут!');
        return;
    }

    // 3. Запуск анализа для всех маршрутов
    const date = document.getElementById('analysisDate').value;
    const results = [];

    for (const route of MultiRouteModule.routes) {
        const result = await RouteModule.analyzeSegments(date);
        results.push(result);
    }

    // 4. Оптимизация распределения
    const weatherData = WeatherModule.cachedData;
    const assignment = MultiRouteModule.optimizeAssignment(weatherData);

    // 5. Отображение результатов
    this.displayOptimizationResults(assignment);

    // 6. Активация кнопки "Далее"
    WizardModule.stepData.segmentAnalysis = results;
    document.querySelector('.step-nav-btn.next').disabled = false;
};
```

canProceed() {

        case 1:
            // Шаг 1: Маршрут + Метеоданные + Точки взлёта

        case 3:
