```javascript
// Псевдокод
function optimizeRoute(start, end, waypoints) {
    const baseRoute = [start, ...waypoints, end];
    const baseRisk = calculateRisk(baseRoute);
    
    // Генерация вариантов
    const alternatives = generateAlternatives(baseRoute);
    
    // Выбор лучшего
    const best = alternatives.reduce((best, alt) => {
        const risk = calculateRisk(alt);
        const distance = calculateDistance(alt);
        const score = risk * 0.7 + distance * 0.3; // Взвешенный
        return score < best.score ? {route: alt, score} : best;
    });
    
    return best.route;
}
```

#### 4. ML-прогнозы (3.1) — 40-60 часов ⭐⭐

**Цель:** Адаптивный выбор лучшего источника прогноза

**Функционал:**
- Сбор фактических данных от пилотов
- Сравнение прогнозов с фактом
- Расчёт точности источников
- Взвешивание прогнозов
- Адаптивная коррекция

**Модель:**
```javascript
// Псевдокод
class ForecastAccuracy {
    constructor() {
        this.history = []; // {source, forecast, actual, error}
    }
    
    addObservation(source, forecast, actual) {
        this.history.push({
            source,
            forecast,
            actual,
            error: Math.abs(forecast - actual),
            date: new Date()
        });
    }
    
    getAccuracy(source, days = 30) {
        const recent = this.history.filter(h => 
            h.source === source && 
            h.date > Date.now() - days*24*60*60*1000
        );
        
        if (recent.length === 0) return 0.5; // Нет данных
        
        const avgError = recent.reduce((sum, h) => sum + h.error, 0) / recent.length;
        return 1 - Math.min(1, avgError / 10); // Нормализация
    }
    
    getBestSource() {
        const sources = ['open-meteo', 'met-no'];
        return sources.reduce((best, source) => {
            const accuracy = this.getAccuracy(source);
            return accuracy > best.accuracy ? {source, accuracy} : best;
        }, {source: 'open-meteo', accuracy: 0});
    }
}
```
