### 3.2.2. Функция getForecast()

**Назначение:** Получение прогноза от Open-Meteo

**Параметры:**
```javascript
{
    lat: number,    // Широта (градусы)
    lon: number,    // Долгота (градусы)
    date: string    // Дата начала (ISO 8601)
```

**Исходный код:**
```javascript
async getForecast(lat, lon, date) {
    const cacheKey = `forecast_${lat}_${lon}_${date}`;
    
    // Проверка кэша
    if (this.cache.has(cacheKey)) {
        const cached = this.cache.get(cacheKey);
        if (Date.now() - cached.timestamp < 3600000) { // 1 час
            Utils.log('Данные загружены из кэша');
            return cached.data;
        }
    }
    
    // Формирование URL
    const params = new URLSearchParams({
        latitude: lat,
        longitude: lon,
        hourly: 'temperature_2m,relative_humidity_2m,dew_point_2m,' +
                'precipitation,rain,showers,snowfall,' +
                'cloud_cover,pressure_msl,wind_speed_10m,' +
                'wind_direction_10m,wind_gusts_10m,visibility',
        daily: 'weather_code,temperature_2m_max,temperature_2m_min,' +
               'precipitation_sum,windspeed_10m_max',
        current_weather: 'true',
        timezone: 'auto',
        forecast_days: '7'
    });
    
    const url = `https://api.open-meteo.com/v1/forecast?${params}`;
    
    try {
        const response = await fetch(url);
        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }
        
        const data = await response.json();
        
        // Кэширование
        this.cache.set(cacheKey, {
            data: data,
            timestamp: Date.now()
        });
        
        Utils.log('Прогноз получен от Open-Meteo');
        return data;
        
    } catch (error) {
        Utils.error('Ошибка получения прогноза:', error);
        throw error;
    }
}
```

### 3.2.3. Функция calculateRiskScore()

**Назначение:** Расчёт балла риска

**Параметры:**
```javascript
{
    hour: object,     // Почасовые данные
    thresholds: object // Пороговые значения
}
```

**Исходный код:**
```javascript
calculateRiskScore(hour, thresholds) {
    let riskScore = 0;
    const details = {};
    
    // Ветер 10м
    if (hour.wind_speed_10m > thresholds.wind10m.medium) {
        riskScore += 2;
        details.wind10m = 'HIGH';
    } else if (hour.wind_speed_10m > thresholds.wind10m.low) {
        riskScore += 1;
        details.wind10m = 'MEDIUM';
    } else {
        details.wind10m = 'LOW';
    }
    
    // Ветер 500м (интерполяция)
    const wind500 = this.interpolateWindForAltitude(hour, 500);
    if (wind500 > thresholds.wind500m.medium) {
        riskScore += 2;
        details.wind500m = 'HIGH';
    } else if (wind500 > thresholds.wind500m.low) {
        riskScore += 1;
        details.wind500m = 'MEDIUM';
    } else {
        details.wind500m = 'LOW';
    }
    
    // Осадки
    if (hour.precipitation > thresholds.precipitation.medium) {
        riskScore += 2;
        details.precipitation = 'HIGH';
    } else if (hour.precipitation > thresholds.precipitation.low) {
        riskScore += 1;
        details.precipitation = 'MEDIUM';
    } else {
        details.precipitation = 'LOW';
    }
    
    // Видимость
    const visibilityKm = (hour.visibility || 10000) / 1000;
    if (visibilityKm < thresholds.visibility.medium) {
        riskScore += 2;
        details.visibility = 'HIGH';
    } else if (visibilityKm < thresholds.visibility.low) {
        riskScore += 1;
        details.visibility = 'MEDIUM';
    } else {
        details.visibility = 'LOW';
    }
    
    // Обледенение
    const icingRisk = this.calculateIcingRisk(hour.temperature_2m, hour.relative_humidity_2m);
    if (icingRisk === 'HIGH') {
        riskScore += 2;
    } else if (icingRisk === 'MEDIUM') {
        riskScore += 1;
    }
    details.icing = icingRisk;
    
    // Турбулентность
    const turbulenceRisk = this.calculateTurbulenceRisk(hour.wind_speed_10m, hour.wind_gusts_10m);
    if (turbulenceRisk === 'HIGH') {
        riskScore += 2;
    } else if (turbulenceRisk === 'MEDIUM') {
        riskScore += 1;
    }
    details.turbulence = turbulenceRisk;
    
    // Ограничение максимума
    riskScore = Math.min(riskScore, 8);
    
    return {
        score: riskScore,
        level: this.getRiskLevel(riskScore),
        details: details
    };
}

getRiskLevel(score) {
    if (score <= 2) return 'LOW';
    if (score <= 5) return 'MEDIUM';
    return 'HIGH';
}
```

### 3.2.4. Функция applyPilotCorrection()

**Назначение:** Коррекция прогноза по фактическим данным

**Исходный код:**
```javascript
applyPilotCorrection(analyzedData, pilotData) {
    const corrected = JSON.parse(JSON.stringify(analyzedData)); // Глубокая копия
    
    // Расчёт коэффициентов коррекции
    const windBias = pilotData.windSpeed / analyzedData.current.wind_speed_10m;
    const tempOffset = pilotData.temperature - analyzedData.current.temperature_2m;
    const humidityBias = pilotData.humidity / analyzedData.current.relative_humidity_2m;
    
    Utils.log(`Коррекция: windBias=${windBias.toFixed(2)}, tempOffset=${tempOffset.toFixed(1)}°C`);
    
    // Применение к почасовым данным
    corrected.hourly = analyzedData.hourly.map(hour => {
        const correctedHour = { ...hour };
        
        // Коррекция ветра (с затуханием для будущих часов)
        const hoursAhead = this.getHoursAhead(hour.time);
        const weight = Math.exp(-hoursAhead / 6); // Затухание за 6 часов
        
        correctedHour.wind_speed_10m = hour.wind_speed_10m * windBias * weight + 
                                        hour.wind_speed_10m * (1 - weight);
        
        // Коррекция температуры
        correctedHour.temperature_2m = hour.temperature_2m + tempOffset * weight;
        
        // Коррекция влажности
        correctedHour.relative_humidity_2m = Math.min(100, hour.relative_humidity_2m * humidityBias);
        
        return correctedHour;
    });
    
    // Принудительная коррекция видимости и облачности
    if (pilotData.visibility < analyzedData.current.visibility / 1000) {
        corrected.current.visibility = pilotData.visibility * 1000;
    }
    
    if (pilotData.cloudBase && pilotData.cloudBase < analyzedData.current.cloud_base) {
        corrected.current.cloud_base = pilotData.cloudBase;
    }
    
    // Пересчёт рисков
    corrected.risk = this.calculateRiskScore(corrected.current, this.thresholds);
    
    // Формирование рекомендаций
    corrected.recommendations = this.generateRecommendations(corrected, pilotData);
    
    return corrected;
}
```

### 3.3.2. Функция segmentRoute()

**Назначение:** Сегментация маршрута на участки ~10 км

**Исходный код:**
```javascript
segmentRoute(points) {
    if (points.length < 2) {
        Utils.error('Маршрут должен содержать минимум 2 точки');
        return [];
    }
    
    const segments = [];
    let currentDistance = 0;
    let segmentStart = points[0];
    let segmentPoints = [points[0]];
    
    for (let i = 1; i < points.length; i++) {
        const dist = this.calculateDistance(
            segmentStart.lat, segmentStart.lon,
            points[i].lat, points[i].lon
        );
        
        currentDistance += dist;
        segmentPoints.push(points[i]);
        
        // Если набралось ~10 км или это последняя точка
        if (currentDistance >= 10 || i === points.length - 1) {
            segments.push({
                start: { ...segmentStart },
                end: { ...points[i] },
                distance: currentDistance,
                points: [...segmentPoints]
            });
            
            // Начало нового сегмента
            segmentStart = points[i];
            segmentPoints = [points[i]];
            currentDistance = 0;
        }
    }
    
    Utils.log(`Маршрут сегментирован: ${segments.length} сегментов`);
    return segments;
}
```

### 3.3.3. Функция calculateDistance() (Haversine)

```javascript
calculateDistance(lat1, lon1, lat2, lon2) {
    const R = 6371; // Радиус Земли, км
    const dLat = (lat2 - lat1) * Math.PI / 180;
    const dLon = (lon2 - lon1) * Math.PI / 180;
    
    const a = 
        Math.sin(dLat / 2) * Math.sin(dLat / 2) +
        Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
        Math.sin(dLon / 2) * Math.sin(dLon / 2);
    
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    
    return R * c;
}
```

```javascript
exportToKML(route) {
    const kmlHeader = `<?xml version="1.0" encoding="UTF-8"?>
<kml xmlns="http://www.opengis.net/kml/2.2">
  <Document>
    <name>${route.name}</name>
    <description>Маршрут БВС</description>
    <Style id="routeStyle">
      <LineStyle>
        <color>ff0000ff</color>
        <width>3</width>
      </LineStyle>
    </Style>`;
    
    // Линия маршрута
    const coordinates = route.points.map(p => 
        `${p.lon},${p.lat},0`
    ).join(' ');
    
    const linePlacemark = `
    <Placemark>
      <name>${route.name}</name>
      <styleUrl>#routeStyle</styleUrl>
      <LineString>
        <tessellate>1</tessellate>
        <coordinates>${coordinates}</coordinates>
      </LineString>
    </Placemark>`;
    
    // Точки
    let pointPlacemarks = '';
    route.points.forEach((point, index) => {
        pointPlacemarks += `
    <Placemark>
      <name>Точка ${String.fromCharCode(65 + index)}</name>
      <Point>
        <coordinates>${point.lon},${point.lat},0</coordinates>
      </Point>
    </Placemark>`;
    });
    
    const kmlFooter = `
  </Document>
</kml>`;
    
    return kmlHeader + linePlacemark + pointPlacemarks + kmlFooter;
}
```

```javascript
async getCurrentLocation() {
    return new Promise((resolve, reject) => {
        if (!navigator.geolocation) {
            reject(new Error('Геолокация не поддерживается браузером'));
            return;
        }
        
        Utils.log('Запрос геолокации...');
        
        navigator.geolocation.getCurrentPosition(
            (position) => {
                const coords = {
                    lat: position.coords.latitude,
                    lon: position.coords.longitude,
                    accuracy: position.coords.accuracy
                };
                Utils.log(`Геолокация получена: ${coords.lat}, ${coords.lon}`);
                resolve(coords);
            },
            (error) => {
                Utils.error('Ошибка геолокации:', error.message);
                reject(error);
            },
            {
                enableHighAccuracy: true,
                timeout: 10000,
                maximumAge: 300000 // 5 минут
            }
        );
    });
}
```

### 3.6.2. Методы хранения

```javascript
save(key, data) {
    const fullKey = this.PREFIX + key;
    const item = {
        data: data,
        timestamp: Date.now()
    };
    localStorage.setItem(fullKey, JSON.stringify(item));
    Utils.log(`Данные сохранены: ${fullKey}`);
}

load(key) {
    const fullKey = this.PREFIX + key;
    const itemStr = localStorage.getItem(fullKey);
    
    if (!itemStr) {
        return null;
    }
    
    try {
        const item = JSON.parse(itemStr);
        return item.data;
    } catch (error) {
        Utils.error('Ошибка загрузки данных:', error);
        return null;
    }
}

remove(key) {
    const fullKey = this.PREFIX + key;
    localStorage.removeItem(fullKey);
    Utils.log(`Данные удалены: ${fullKey}`);
}

clearExpiredCache() {
    const now = Date.now();
    const cacheKey = this.PREFIX + 'cache_weather';
    const item = localStorage.getItem(cacheKey);
    
    if (item) {
        try {
            const parsed = JSON.parse(item);
            if (now - parsed.timestamp > this.CACHE_TTL) {
                localStorage.removeItem(cacheKey);
                Utils.log('Кэш очищен (истёк срок)');
            }
        } catch (e) {
            localStorage.removeItem(cacheKey);
        }
    }
}
```

```javascript
async function analyzeRoute(route) {
    // Получение прогноза для центральной точки
    const centerLat = route.points.reduce((s, p) => s + p.lat, 0) / route.points.length;
    const centerLon = route.points.reduce((s, p) => s + p.lon, 0) / route.points.length;
    
    const forecast = await WeatherModule.getForecast(centerLat, centerLon, new Date());
    
    // Анализ
    const analysis = WeatherModule.analyzeForecast(forecast);
    
    // Поиск благоприятных окон
    const windows = WeatherModule.findFlightWindows(forecast.hourly, WeatherModule.thresholds);
    
    // Рекомендации
    const recommendations = WeatherModule.generateRecommendations(analysis, null);
    
    return { forecast, analysis, windows, recommendations };
}
```
